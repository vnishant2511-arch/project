import React from "react";

// Base Person class
class Person {
  constructor(name, age) {
    this.name = name;
    this.age = age;
  }

  getInfo() {
    return `Name: ${this.name}, Age: ${this.age}`;
  }
}

// Student subclass
class Student extends Person {
  constructor(name, age, course) {
    super(name, age); // call parent constructor
    this.course = course;
  }

  // override method
  getInfo() {
    return `${super.getInfo()}, Course: ${this.course}`;
  }
}

// Teacher subclass
class Teacher extends Person {
  constructor(name, age, subject) {
    super(name, age);
    this.subject = subject;
  }

  // override method
  getInfo() {
    return `${super.getInfo()}, Subject: ${this.subject}`;
  }
}

// React component to demonstrate
const App = () => {
  const student = new Student("Alice", 20, "Computer Science");
  const teacher = new Teacher("Mr. John", 40, "Mathematics");

        <h3>Teacher</h3>

};
